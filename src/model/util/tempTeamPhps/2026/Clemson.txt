function addZeroToHex(d,h) {
    if (d<16) h = "0" + h;
    return h.substr(0,2);
}

function getColor(rank,teams,games) {
    // if (teams>1 && games>0) {var decimalColor = (rank-1)/(teams-1);}
    if (teams>1) {var decimalColor = (rank-1)/(teams-1);}
    else {var decimalColor = 0.5;}
    if (decimalColor<=.5) {
        var fraction = Math.pow((1-(.5-decimalColor)/.5),.2);
        var red = 94.9*fraction;
        var green = (1-fraction)*1.96+98.04;
        var blue = 99.22*fraction;
    }
    else {
        var fraction = Math.pow((1+(.5-decimalColor)/.5),.2);
        var red = (1-fraction)*5.1+94.9;
        var green = 98.84*fraction;
        var blue = fraction*99.22;
    }
    red=2.55*red;
    green=2.55*green;
    blue=2.55*blue;
    var hexRed = dechex(red);
    var hexGreen = dechex(green);
    var hexBlue = dechex(blue);
    hexRed = addZeroToHex(red,hexRed);
    hexGreen = addZeroToHex(green,hexGreen);
    hexBlue = addZeroToHex(blue,hexBlue);
    var bgColor = "#"+hexRed+hexGreen+hexBlue

    return bgColor;
}
